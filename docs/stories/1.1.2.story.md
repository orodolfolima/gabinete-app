## 📝 Implementação

```typescript
// backend/src/routes/visitantes.ts
import { Router } from 'express';
import { authenticateJWT } from '../middleware/auth';
import { validateCPF, validateEmail } from '../utils/validators';

const router = Router();

router.post('/', authenticateJWT, async (req, res) => {
  const { cpf, nome, email, endereco } = req.body;

  // Validações
  if (!validateCPF(cpf)) return res.status(400).json({ error: 'CPF inválido' });
  if (!validateEmail(email)) return res.status(400).json({ error: 'Email inválido' });

  const visitante = await prisma.visitante.create({
    data: {
      cpf,
      nome,
      email,
      endereco: { create: endereco }
    }
  });

  res.status(201).json(visitante);
});

router.get('/', authenticateJWT, async (req, res) => {
  const { categoria, limite = 20, offset = 0 } = req.query;

  const where = categoria ? { categoria: String(categoria) } : {};

  const [data, total] = await Promise.all([
    prisma.visitante.findMany({
      where,
      take: Number(limite),
      skip: Number(offset),
      include: { endereco: true }
    }),
    prisma.visitante.count({ where })
  ]);

  res.json({
    data,
    total,
    limite: Number(limite),
    offset: Number(offset),
    hasMore: offset + Number(limite) < total
  });
});

export default router;
```

describe('Visitantes API', () => {

  });
});
